# Login ECR
aws ecr get-login-password --region us-east-1 \
| docker login --username AWS --password-stdin <ACCOUNT_ID>.dkr.ecr.us-east-1.amazonaws.com

# Build web
docker build -t chroma-chat-web:latest 03-ia/app
docker tag chroma-chat-web:latest <ACCOUNT_ID>.dkr.ecr.us-east-1.amazonaws.com/chroma-chat-web:latest
docker push <ACCOUNT_ID>.dkr.ecr.us-east-1.amazonaws.com/chroma-chat-web:latest

# Build ingest
docker build -t chroma-ingest:latest 03-ia/ingest
docker tag chroma-ingest:latest <ACCOUNT_ID>.dkr.ecr.us-east-1.amazonaws.com/chroma-ingest:latest
docker push <ACCOUNT_ID>.dkr.ecr.us-east-1.amazonaws.com/chroma-ingest:latest
